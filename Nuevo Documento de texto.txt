    const idUsuario = localStorage.getItem("id_usuario");

    if (!idUsuario) {
      alert("‚ö†Ô∏è No has iniciado sesi√≥n.");
      window.location.href = "login.html";
    }

    async function cargarCupones() {
      try {
        const res = await fetch(`php/obtener_cupones.php?id_usuario=${idUsuario}`);
        const data = await res.json();

        const lista = document.getElementById("listaCupones");
        lista.innerHTML = "";

        if (data.status === "ok" && data.cupones.length > 0) {
          data.cupones.forEach(c => {
            const estadoClass = c.usado == 1 ? "estado-usado" : "";
            const qrURL = `https://api.qrserver.com/v1/create-qr-code/?size=150x150&data=${encodeURIComponent(c.codigo_unico)}`;

            lista.innerHTML += `
              <div class="cupon ${estadoClass}">
                <div class="cupon-info">
                  <h2>${c.nombre}</h2>
                  <p>${c.descripcion}</p>
                  <p><strong>Tipo:</strong> ${c.tipo}</p>
                  <p><strong>C√≥digo:</strong> ${c.codigo_unico}</p>
                </div>
                <img src="${qrURL}" class="qr" alt="QR de cup√≥n">
              </div>
            `;
          });
        } else {
          lista.innerHTML = "<p>No ten√©s cupones asignados todav√≠a.</p>";
        }

      } catch (error) {
        alert("‚ö†Ô∏è Error al cargar cupones.");
      }
    }

    cargarCupones();
  </script>
